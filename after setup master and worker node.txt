vi deployment.yml

 apiVersion: apps/v1
kind: Deployment
metadata:
  name: node-app-deployment
  
  labels:
    app: node-app
spec:
  replicas: 2
  selector:
    matchLabels:
      app: node-app
  template:
    metadata:
      name: node-pod
      namespace: node-app
      labels:
        app: node-app
    spec:
      containers:
        - name: node-container
          image: trainwithshubham/node-app-batch-6
          ports:
            - containerPort: 8000
          resources:
            requests:
              memory: "64Mi"
              cpu: "250m"
            limits:
              memory: "128Mi"
              cpu: "500m"
:wq

kubectl apply -f deployment.yml
